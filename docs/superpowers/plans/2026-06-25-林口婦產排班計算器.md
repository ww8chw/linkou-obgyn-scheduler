> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 建立一個單一 HTML 檔的純前端工具，輸入當月參數後計算林口長庚婦產科住院醫師各班別/各職級/每人應值的平假班數。

**Architecture:** 計算邏輯先寫成獨立的 `scheduler.js`（純函式、可用 Node 測試），台灣 2026 假日資料放 `holidays2026.js`，兩者以 TDD 開發。最後 `build.js` 把它們內聯進 `index.html`，產出可離線使用的單檔。

**Tech Stack:** Vanilla JS (ES modules for dev/test), Node.js 內建 `node:test` + `node:assert`（無外部相依），最終單一 HTML 檔。

## File Structure

- `src/holidays2026.js` — 2026 台灣官方假日/補班資料 + `getMonthDays(year, month)` 算平假天數與最後一天屬性。
- `src/scheduler.js` — 核心演算法：`calculateSchedule(input)` 回傳結構化結果。內含各班別填補、特休、上限、單調性邏輯，拆成小的可測純函式。

- `src/ui.js` — 讀表單 → 呼叫 `calculateSchedule` → 渲染結果與警示（不可單元測，靠手動驗收）。

- `build.js` — 把 holidays2026.js / scheduler.js / ui.js 內聯進 template，輸出 `index.html`。
- `index.html` — 最終產物（build 產生）。

```js
export function getMonthDays(year, month) {
  const holidaySet = new Set(HOLIDAYS_2026);
  const makeupSet = new Set(MAKEUPS_2026);
  const daysInMonth = new Date(year, month, 0).getDate();
  let weekday = 0, weekend = 0;
  let lastDayIsWeekend = false;
  for (let d = 1; d <= daysInMonth; d++) {
    const date = new Date(year, month - 1, d);
    const iso = `${year}-${String(month).padStart(2,'0')}-${String(d).padStart(2,'0')}`;
    const dow = date.getDay(); // 0=日,6=六
    let isHoliday = (dow === 0 || dow === 6);
    if (makeupSet.has(iso)) isHoliday = false;   // 補班日算平日
    if (holidaySet.has(iso)) isHoliday = true;    // 放假日算假日
    if (isHoliday) weekend++; else weekday++;
    if (d === daysInMonth) lastDayIsWeekend = isHoliday;
  }
  return { weekday, weekend, lastDayIsWeekend };
}
```

```js
const D = (v) => (v == null ? 0 : Number(v));
export function normalizeInput(raw) {
  const lvl = (o = {}) => ({ ...o, count: D(o.count) });
  return {
    month: { weekday: D(raw.month.weekday), weekend: D(raw.month.weekend), lastDayIsWeekend: !!raw.month.lastDayIsWeekend },
```

```js
export const MAX_TOTAL = 8, MAX_WEEKEND = 3, MAX_WEEKDAY = 5; // 上限：總8、假3、平5(6平2假亦可故平日彈性見capWeekday)

export function createPersonPool(levels) {
  const pool = {};
  for (const [name, lv] of Object.entries(levels)) {
    pool[name] = {
      count: lv.count, weekday: 0, weekend: 0,
      bigLeave: lv.bigLeave || 0, smallLeave: lv.smallLeave || 0,
      keelung: lv.keelung || 0,
      byShift: {}, // {L1:{weekday,weekend},...}
    };
  }
  return pool;
}

// 群組剩餘可吃容量（以人數×上限估算，平日上限取6以容許6平2假）
function groupCapacity(g) {
  const wkCap = g.count * 6 - g.weekday;          // 平日寬鬆到6
  const wendCap = g.count * MAX_WEEKEND - g.weekend;
  const totalCap = g.count * MAX_TOTAL - (g.weekday + g.weekend);
  return { wkCap: Math.max(0, wkCap), wendCap: Math.max(0, wendCap), totalCap: Math.max(0, totalCap) };
}

function addToGroup(g, shift, wk, wend) {
  g.weekday += wk; g.weekend += wend;
  g.byShift[shift] = g.byShift[shift] || { weekday: 0, weekend: 0 };
  g.byShift[shift].weekday += wk; g.byShift[shift].weekend += wend;
}

export function fillShift(pool, order, shift, demand, opts = {}) {
  let { weekday, weekend } = demand;
  for (const name of order) {
    const g = pool[name];
    if (!g || g.count === 0) continue;
    let cap = groupCapacity(g);
    // 假日優先吃（稀缺），但不可超 totalCap
    let takeWend = Math.min(weekend, cap.wendCap, cap.totalCap);
    let takeWk = Math.min(weekday, cap.wkCap, cap.totalCap - takeWend);
    // 自訂上限（如 R3 L班 班數上限）
    if (opts.shiftCap != null) {
      const room = opts.shiftCap - ((g.byShift[shift]?.weekday||0)+(g.byShift[shift]?.weekend||0));
      const allow = Math.max(0, room);
      const both = takeWk + takeWend;
      if (both > allow) { // 等比例縮減，先保假日
        takeWend = Math.min(takeWend, allow);
        takeWk = Math.min(takeWk, allow - takeWend);
      }
    }
    if (takeWk + takeWend > 0) addToGroup(g, shift, takeWk, takeWend);
    weekday -= takeWk; weekend -= takeWend;
    if (weekday <= 0 && weekend <= 0) break;
  }
  return { weekday: Math.max(0, weekday), weekend: Math.max(0, weekend) };
}
```

```js
export function calculateSchedule(raw) {
  const n = normalizeInput(raw);
  const demands = buildDemands(n);
  const L = n.linko, T = n.taipei;

  // 林口人員池
  const linko = createPersonPool({
    y2: L.y2, r1: L.r1, r2: L.r2, r3: L.r3, r4: L.r4, f1: L.f1, f2: L.f2, f3: L.f3,
  });
  // 台北人員池
  const taipei = createPersonPool({ tr1to3: { count: T.r1to3 }, tr4: { count: T.r4 }, tf1: { count: T.f1 } });

  const warnings = [];
  const leftover = (label, r) => { if (r.weekday > 0 || r.weekend > 0) warnings.push(`${label} 缺人：尚有 ${r.weekday}平${r.weekend}假 無法分配`); };

  // 1 病房班：Y2(可值最後一天)→R1→R2→R3
  leftover('病房班', fillShift(linko, ['y2','r1','r2','r3'], 'ward', demands.ward));
  // 2 L5：Y2、R1 → R2→R3→R4→F1
  leftover('L5', fillShift(linko, ['y2','r1','r2','r3','r4','f1'], 'L5', demands.L5));
  // 3 L1：R2→R3
  leftover('L1', fillShift(linko, ['r2','r3'], 'L1', demands.L1));
  // 4 L：R3(受班數上限)→R4→F1→F2
  {
    let rem = fillShift(linko, ['r3'], 'L', demands.L, { shiftCap: L.r3.lCapShifts || 0 });
    rem = fillShift(linko, ['r4','f1','f2'], 'L', rem);
    leftover('L', rem);
  }
  // 5 T1：台北R1-3→台北R4→林R1→林R2→林R3→林R4→台北F1（基隆籍跳過，由createPersonPool keelung控制，見Task6）
  {
    let rem = fillShift(taipei, ['tr1to3','tr4'], 'T1', demands.T1);
    rem = fillShift(linko, ['r1','r2','r3','r4'], 'T1', rem, { excludeKeelung: true });
    rem = fillShift(taipei, ['tf1'], 'T1', rem);
    leftover('T1', rem);
  }
  // 6 T2：台北F1→林R4→林F1→林F2→林F3
  {
    let rem = fillShift(taipei, ['tf1'], 'T2', demands.T2);
    rem = fillShift(linko, ['r4','f1','f2','f3'], 'T2', rem, { excludeKeelung: true });
    leftover('T2', rem);
  }
  // 7 LR：林R4→林F1→林F2→林F3
  leftover('LR', fillShift(linko, ['r4','f1','f2','f3'], 'LR', demands.LR));

  return formatResult(linko, taipei, n, warnings);
}
```

```js
// 將群組總班數依特休拆解為 normal/big/small 每人平假，並做同級吸收
export function applyLeaveAdjustment(g) {
  const { count, weekday, weekend, bigLeave = 0, smallLeave = 0 } = g;
  const normalCount = count - bigLeave - smallLeave;
  // 假日：所有人平均（不因特休改變）
  const wendEach = count > 0 ? Math.round(weekend / count) : 0;
  // 平日基準（無特休者）：先估均值
  const baseWk = count > 0 ? Math.floor(weekday / count) : 0;
  const big = { weekday: Math.max(0, baseWk - 2), weekend: wendEach };
  const small = { weekday: Math.max(0, baseWk - 1), weekend: wendEach };
  const normal = { weekday: baseWk, weekend: wendEach };
  // 特休少值的平日總額
  const freed = bigLeave * 2 + smallLeave * 1;
  // 同級無特休者吸收
  let normalAdjusted = { ...normal };
  let pushedUp = { weekday: 0, weekend: 0 };
  if (normalCount > 0) {
    const addEach = Math.floor(freed / normalCount);
    const newWk = baseWk + addEach;
    if (newWk > MAX_TOTAL) { // 超8 → 往上推
      normalAdjusted.weekday = MAX_TOTAL - wendEach;
      pushedUp.weekday = (newWk - (MAX_TOTAL - wendEach)) * normalCount;
    } else {
      normalAdjusted.weekday = newWk;
    }
  } else {
    pushedUp.weekday = freed; // 全員特休，整批往上推
  }
  return { normal, normalAdjusted, big, small, pushedUp, normalCount, bigLeave, smallLeave };
}
```

```js
function validateLimits(linko, warnings) {
  const minF = { f1: 3, f2: 2, f3: 1 };
  let anyFbelow = false;
  for (const [k, min] of Object.entries(minF)) {
    const g = linko[k];
    if (g.count > 0) {
      const each = (g.weekday + g.weekend) / g.count;
      if (each < min) { anyFbelow = true; warnings.push(`${k.toUpperCase()} 每人 ${each.toFixed(1)} 班，低於最低 ${min} 班`); }
    }
  }
  for (const k of ['y2','r1','r2','r3','r4','f1','f2','f3']) {
    const g = linko[k];
    if (g.count === 0) continue;
    const totEach = (g.weekday + g.weekend) / g.count;
    const wendEach = g.weekend / g.count;
    if (totEach > MAX_TOTAL + 1e-9) warnings.push(`${k.toUpperCase()} 每人約 ${totEach.toFixed(1)} 班，超過 8 班上限`);
    if (wendEach > MAX_WEEKEND + 1e-9) warnings.push(`${k.toUpperCase()} 每人約 ${wendEach.toFixed(1)} 假日班，超過 3 假上限`);
    if (!anyFbelow && ['r1','r2','r3','r4'].includes(k) && totEach < MAX_TOTAL - 1e-9) {
      warnings.push(`${k.toUpperCase()} 每人約 ${totEach.toFixed(1)} 班，低於 8 班（無 F 級缺額時 R 級不應低於 8）`);
    }
  }
}
```
